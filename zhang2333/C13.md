```cpp
#include<cstdio>
#include<cstring>
#define xxx 5202333
using namespace std;
char s[xxx];
int len,p;
long long l,r;

int main()
{
	scanf("%s",s+1);
	len=strlen(s+1);
	for (int i=1;i<=len;i++)
		if (s[i]=='^')
			p=i;
	for (long long i=p-1;i>0;i--)
	{
		if (s[i]>='1'&&s[i]<='9')
		{
			l+=(long long)(p-i)*(s[i]-'0');
		}
	}
	for (long long i=p+1;i<=len;i++)
	{
		if (s[i]>='1'&&s[i]<='9')
		{
			r+=(long long)(i-p)*(s[i]-'0');
		}
	}
	if (l==r) printf("balance");
	else if (l>r) printf("left");
	else printf("right");
	return 0;
}
```

```cpp
#include<cstdio>

int n,m,tot[xxx]={0};

int abss(int x)
{
	return x>0?x:-x;
}

int main()
{
	int a,b,c,ans;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++)
	{
		scanf("%d%d%d",&a,&b,&c);
		tot[a]-=c;
		tot[b]+=c;
	}
	ans=0;
	for (int i=1;i<=n;i++)
	{
		ans+=abss(tot[i]);
	}
	ans/=2;
	printf("%d",ans);
	return 0;
}
```
